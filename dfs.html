C++ DFS Graph Traversal

This webpage showcases the implementation of Depth-First Search (DFS) for graph traversal in C++. Below is the code:

#include <iostream>
#include <vector>
using namespace std;

class Graph {
public:
    int vertices;
    vector<vector<int>> adjList;

    Graph(int v) {
        vertices = v;
        adjList.resize(v);
    }

    // Add an edge to the graph (undirected)
    void addEdge(int u, int v) {
        adjList[u].push_back(v);
        adjList[v].push_back(u);
    }

    // Recursive DFS function
    void DFS(int node, vector<bool>& visited) {
        // Mark the node as visited and print it
        visited[node] = true;
        cout << node << " ";

        // Visit all unvisited neighbors of the node
        for (int neighbor : adjList[node]) {
            if (!visited[neighbor]) {
                DFS(neighbor, visited);
            }
        }
    }

    // Function to call DFS from a given start node
    void DFSTraversal(int start) {
        vector<bool> visited(vertices, false);
        DFS(start, visited);
    }
};

int main() {
    Graph g(6); // Create a graph with 6 nodes (0 to 5)

    // Adding edges to the graph
    g.addEdge(0, 1);
    g.addEdge(0, 2);
    g.addEdge(1, 3);
    g.addEdge(1, 4);
    g.addEdge(2, 5);

    cout << "DFS traversal starting from node 0: ";
    g.DFSTraversal(0);

    return 0;
}

C++ Code: Bubble Sort

This is a C++ implementation of the Bubble Sort algorithm. It sorts an array of integers in ascending order.

Click the button below to see how Bubble Sort works on a sample array.

#include <iostream> using namespace std; // Function to perform Bubble Sort void bubbleSort(int arr[], int n) { // Outer loop for each pass through the array for (int i = 0; i < n - 1; i++) { // Inner loop for comparing adjacent elements for (int j = 0; j < n - i - 1; j++) { // If the element is greater than the next element, swap them if (arr[j] > arr[j + 1]) { swap(arr[j], arr[j + 1]); } } } } // Function to print the array void printArray(int arr[], int n) { for (int i = 0; i < n; i++) { cout << arr[i] << " "; } cout << endl; } int main() { int arr[] = {64, 34, 25, 12, 22, 11, 90}; int n = sizeof(arr) / sizeof(arr[0]); cout << "Original Array: "; printArray(arr, n); bubbleSort(arr, n); // Calling Bubble Sort cout << "Sorted Array: "; printArray(arr, n); // Printing the sorted array return 0; }
